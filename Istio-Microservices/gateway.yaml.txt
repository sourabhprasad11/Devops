apiVersion: networking.istio.io/v1alpha3
kind: Gateway
metadata:
  name: chatapp-gateway
spec:
  selector:
    istio: ingressgateway # use istio default controller
  servers:
  - port:
      number: 80
      name: http
      protocol: HTTP
    hosts:
    - "*"
---
apiVersion: networking.istio.io/v1alpha3
kind: VirtualService
metadata:
  name: chatapp
spec:
  hosts:
  - "*"
  gateways:
  - chatapp-gateway
  http:
  - match:
    - uri:
        exact: /
    - uri:
        prefix: /static
    - uri:
        exact: /login
    - uri:
        exact: /logout
    - uri:
        exact: /register
    route:
    - destination:
        host: ex.south-1a.tk/ 
        port:
          number: 9080
